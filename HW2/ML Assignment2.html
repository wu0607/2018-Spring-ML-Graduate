class Gaussian_feature():
    def __init__(self, O1, O2, x1_max=None, x1_min=None, x2_max=None, x2_min=None):
        self.O1 = O1
        self.O2 = O2
        self.P = O1 * O2
        self.s1 = (x1_max - x1_min) / (O1-1)
        self.s2 = (x2_max - x2_min) / (O2-1)
    
    def get_feature(self, x):
        N = x.shape[0]
        sea_indicator = x[:, 2].astype(np.float32) # sea indicator 
        x_i = x[:, 0]
        x_j = x[:, 1]
        phi_x = np.ones((N, self.P+2))   # (N, P+2)
        phi_x[:, self.P] = sea_indicator # P+1 position is see indicator
        for n in range(N):  # For each x point take x_i, x_j 2 features transfer to P gaussian features
            for i in range(1, self.O1+1, 1):
                for j in range(1, self.O2+1, 1):
                    k = self.O2 * (i-1) + j
                    mu_i = self.s1 * (i-1)
                    mu_j = self.s2 * (j-1)
                    phi_x[n, k-1] = np.exp( - np.square(x_i[n] - mu_i)/(2*np.square(self.s1)) \
                                            - np.square(x_j[n] - mu_j)/(2*np.square(self.s2)) )
        
        return phi_x

Linear Regression¶

We have closed form solution for weight: $\hat{w} = (X^TX)^{-1}X^Ty$

In [7]:
from numpy.linalg import pinv # use pseudo-inverse to avoid singular matrix occur

class linear_regression():
    
    def __init__(self):
        self.weight = None
    
    def fit(self, X_train, y_train):
        X = np.array(X_train)
        y = np.array(y_train)
    
        product = np.dot(X.T, X)
        theInverse = pinv(product)
        self.weight = np.dot(np.dot(theInverse, X.T), y)
        
        return self

    def predict(self, X):
        return np.dot(X, self.weight)
    
def SSE(prediction , ground_truth):
    return np.sum(np.square(prediction - ground_truth))

class Bayesian_regression():
    
    def __init__(self, beta=1):
        # Here we assume beta equal 1, to make it simple
        self.weight = None
        self.beta = beta
    
    def fit(self, X_train, y_train):
        N, dim = X_train.shape
        # prior 
        m_0 = np.zeros((dim, 1)) # zero mean
        S_0 = np.eye(dim)        # identical matrix
        S_0_inv = pinv(S_0)
        
        # posterior
        SN_inv = pinv(S_0) + self.beta * np.dot(X_train.T, X_train)
        SN = pinv(SN_inv)
        mN = np.dot(SN, (np.dot(S_0_inv, m_0) + self.beta * np.dot(X_train.T, y_train)))
        
        self.weight = mN 
        return self

    def predict(self, X):
        return np.dot(X, self.weight)
